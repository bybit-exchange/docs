Get launched instruments information.

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

### HTTP Request
GET `/derivatives/v3/public/instruments-info`

### Request Parameters
| Parameter | Required | Type | Comments|
|:----- |:-------|:-----|------ |
|[category](../enum#category) |false |string |Product type. `linear`,`inverse`,`option`. default: `linear`, but in the response category shows `""`|
|symbol |false |string |Symbol name. <li>query an option symbol, `category` is **required**</li>|
|baseCoin |false |string |Base coin. <li>Valid when `category`=option</li><li>BTC by default if not passed</li>|
|limit |false |integer |Limit for data size per page. [1,1000]. default: 500|
|cursor |false |string |Cursor. Used to pagination|
